Option Explicit 'Enforces variable declaration, reducing errors from misspelled variable names

Dim intHeight As Double
Dim intWidth As Double
Dim intArea As Double
Dim strText As String
Dim strText2 As String
Dim boolConfirm As Boolean
Dim intForItem As Integer

Sub main()

    intHeight = 1
    intWidth = 5.1254
    intArea = intHeight * intWidth
    strText = "LABAS!!!<---"
    strText2 = "VISO!!!<---"
    
    MsgBox "Area: " & intArea
    
    boolConfirm = False
    
    If boolConfirm = True Or intArea < 30 Then
        MsgBox strText
    Else
        For intForItem = 1 To 3
            MsgBox "Number: " & intForItem
        Next intForItem
        
        MsgBox strText2
    End If
    
End Sub
